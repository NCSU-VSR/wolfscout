#!/bin/bash -ex
if [ -d "ncsu" ]; then
    echo "**> virtualenv exists"
    echo "**> virtualenv ncsu is being deleted"
    rm -rf ncsu
else
    echo "**> creating virtualenv"
fi
virtualenv --no-site-packages ncsu
source ./ncsu/bin/activate
export PYTHONPATH=.:$WORKSPACE
export DJANGO_SETTINGS_MODULE=settings.testSettings
pip install -E ./ncsu -r requirements.txt
django-admin.py test
