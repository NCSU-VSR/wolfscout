#!/bin/bash -ex
cd ../..
virtualenv --no-site-packages ncsu
cd ncsu
source ./bin/activate
cd wolfscout
export PYTHONPATH=.:/opt/webapps/hudson/ncsu
export DJANGO_SETTINGS_MODULE=wolfscout.settings.testSettings
pip install -q -r requirements.txt
django-admin.py test --with-xcoverage --cover-package=apps.crawler --with-xunit
